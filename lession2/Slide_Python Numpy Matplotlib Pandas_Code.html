# 3. Vit hm tnh ton
def one_day(myNums, result, money):
    pay = len(myNums) * money
    get = money * 70 if result in myNums else 0
    return get-pay

def many_day(myNums, results, money):
    total = 0
    for x in results:
        total += one_day(myNums, x, money)
    return total
